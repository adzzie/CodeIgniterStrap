Launch demo modal 
<?php echo Tb::modal(array(
    'id' => 'myModal',
    'header' => 'Modal Heading',
    'body' => '<p>One fine body...</p>',
    'footer' => array(
        Tb::button('Save Changes', array('data-dismiss' => 'modal', 'color' => Tb::BUTTON_COLOR_PRIMARY)),
        Tb::button('Close', array('data-dismiss' => 'modal')),
     ),
)); ?>


<?php echo Tb::button('Launch demo modal', 
    array('type' => Tb::BUTTON_TYPE_LINK, 'url' => '#myModal', 'data-toggle' => 'modal')); ?>

)); ?>
